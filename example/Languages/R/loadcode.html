aardio 范例: 调用 R 语言 - 嵌入模板

//aardio 调用 R 语言 - 嵌入模板
import console.int; 
import process.r;

//执行 R 代码，支持模板语法： 
//https://www.aardio.com/zh-cn/doc/language-reference/templating/syntax.html
var prcs = process.r.loadcode(`write("<?

//可以在 R 代码中嵌入 aardio  代码
if(_WIN10_LATER){
    print("新系统",owner.模板参数名)
}
else {
    print("旧系统",owner.模板参数名)    
}


?>",file=".data.txt");`,{
    模板参数名 = "参数值"
})

prcs.logResponse();

console.log(string.load("/.data.txt"))
