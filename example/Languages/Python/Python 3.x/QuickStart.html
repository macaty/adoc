aardio 范例: 调用 Python 快速入门

//aardio 调用 Python 快速入门
import console;  
import py3;

//导入 Python 模块
var math = py3.import("math");

//调用 Python 函数，  
var num = math.floor(22.3);

//在控制台输出数值。 
console.log(num);

/*
请参考 Python 数据类型转换规则: 
https://www.aardio.com/zh-cn/doc/library-guide/ext/python/conversion.html
*/

console.pause();
