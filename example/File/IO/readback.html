aardio 范例: 倒读文件 

//倒读文件 
import console;

//打开文件 https://www.aardio.com/zh-cn/doc/library-guide/builtin/io/file.html
file = io.file("~\Config\intellisense\kernel.txt","r+");

//移动文件指针到尾部
file.seek("end") 

//读取最后一行
console.log( file.readback() )

//读取倒数第二行
console.log( file.readback() )

//读取倒数第三行
console.log( file.readback() )

//再向前读取12个字节
console.log( file.readback(12) )

//向前读取到文件头部第二个字节
//console.log( file.readback(-2) ) 
console.pause();
