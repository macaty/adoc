aardio 范例: zip 解压

//zip 解压
import console;
import fsys.dlg; 
import zlib.unzip;

var ok,err = zlib.unzip.extract( fsys.dlg.open("*.zip|*.zip","请选择需要解压的zip文件"),"/my",
    function(fileName,extractPath,fileInfo,size,unitSize,unitName){
        if( size !== null )
            console.log( "正在解压文件",fileName,size++ unitName );
        else {
            console.log( "正在解压目录",fileName );
        }

        return true;
    }
)

console.pause();
