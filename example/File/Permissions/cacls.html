aardio 范例: 修改文件权限

//修改文件权限
import fsys;
import process.popen;
import sys.acl;

var path = io.fullpath( "/cacls.aardio" ) 

//禁止所有文件权限,禁止删除
var prcs = process.popen("cacls.exe",path,"/P ",sys.acl.getUserName()+":n", "/C")
prcs.write('y\r\n');
prcs.close();

//恢复完全控制权限
