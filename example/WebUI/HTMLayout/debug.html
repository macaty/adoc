)
/*}}*/

import web.layout;
wbLayout = web.layout(winform.layoutWindow);

***/

 //小技巧: 输入 wbLayout.debug() 可自动完成下面的代码

wbLayout.attachEventHandler( web.layout.debug ); //这句代码为CSSS!增加了全局函数 debug()

wbLayout.css = /**
#my-button{ 
  assigned!:
    debug("CSS脚本中调用debug函数","输出一个或多个值")
  ; 
}
**/

wbLayout.css = /**
#my-button{  
  height 19px; //这里制造了一个CSS语法错误
}
**/
