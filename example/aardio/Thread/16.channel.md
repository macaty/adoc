# aardio 渚: 绾跨

```aardio aardio
//绾跨
import console;
import thread.channel;

//寤洪
var chan = thread.channel(10);

//寤烘扮绾跨
thread.invoke(
    function(chan){

        //煎伴锛濡缂瀛婊′灏辩寰?        chan <<= "杩瀛绗涓?";
        chan <<= "杩瀛绗涓?";
        chan <<= "杩瀛绗涓?";
        chan <<= "杩瀛绗涓?";

    },chan
)

//濡绌猴浠朵绾跨ㄨラ锛寰璇诲
while( ! chan.empty() ){
    /*
    娉ㄦ锛充娇缂瀛奸绌轰娌℃版锛
    濡浠朵寮ㄨラ绾跨ㄥ伐浣涓锛 chan.empty() 浠朵杩?false ?    浠峰涓奸垮害圭 chan.len() 芥般?    */
    var v;

    //璇诲硷娌℃煎氨绛寰
    v <<= chan;

    console.log(v);
}

//涔浠ヨ峰锛寰璇诲缂瀛硷存ラ涓虹┖
for v in chan.each(){
    console.log(v);
}

console.pause(true);

```
