# aardio 渚: 绾跨娑

```aardio aardio
//绾跨娑
import win;
import console;

var thrdFunc = function(arg){

    //绾跨芥板ㄨ娣诲宸茬import璇
    import win;
    import console;

    //ㄥ绾跨ㄦ寰
    win.loopMessage(
        //娉ㄥ涓涓娑╁芥?        function(msg){
            console.log( "跺版",msg.message )
        }
    )

}

//寤虹嚎绋
var thrdHandle,thrdId = thread.create( thrdFunc  )
sleep(1000)

//缁绾跨娑?::PostThreadMessage(thrdId,123,0,0)
::PostThreadMessage(thrdId,456,0,0)

//缁绾跨烘?::PostThreadMessage(thrdId,0x12/*_WM_QUIT*/,0,0)

thread.waitClose(thrdHandle);
console.log("绾跨宸查?)

execute("pause") //浠绘缁х画

```
