# aardio 渚: 璁℃颁俊烽姝?
```aardio aardio
//璁℃颁俊烽姝?import console;
import thread.semaphore;
var semaphore = thread.semaphore(,2);

var workThread = function(semaphore){
    import console;

    console.log('ㄧ寰轰') sleep(1)
    semaphore.wait();

    console.log("轰ヤ,姝ｅㄥㄨ婧?.....")
    sleep(2000);

    semaphore.release()

    console.log("宸查捐婧?.....") sleep(1)
}

console.log("轰涓や釜,浣浣浠芥宠?......")
console.log("---------------")

import thread.manage
manage = thread.manage()

//寤10涓绾跨?for(i=1;10;1){
    manage.create( workThread,semaphore )
    sleep(100)
}

manage.waitClose();

console.log("---------------")
console.log("涓垮缁浜璧板浜?......")
execute("pause")

```
