aardio 范例: 分步请求

//分步请求
import console;
import inet.http;

//首先创建 HTTP 客户端对象
var http = inet.http();

//创建 HTTP 请求，参数 @2指定请求方法，更多参数请查看函数说明
http.beginRequest("http://www.aardio.com","GET");

//发送请求
http.send();

//读取 HTTP 响应头（要在发送请求头后才能读取 ）。
var headers = http.readHeader();
console.log(headers);

//读取数据
var data = http.readAll();
console.log(data);
console.pause()
